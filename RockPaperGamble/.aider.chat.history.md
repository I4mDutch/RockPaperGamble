import { useEffect, useState } from "react";
import { GameSettings } from "../../shared";

type Props = {
  isOpen: boolean;
  onClose: () => void;
  settings: GameSettings;
  onApply: (next: GameSettings) => void;
};

export function SettingsModal({ isOpen, onClose, settings, onApply }: Props) {
  const [form, setForm] = useState<GameSettings>(settings);

  // Keep form in strict 1:1 sync with incoming settings
  useEffect(() => {
    setForm(settings);
  }, [settings]);

  if (!isOpen) return null;

  const update = <K extends keyof GameSettings>(key: K, value: GameSettings[K]) => {
    setForm((prev) => ({ ...prev, [key]: value }));
    onApply({ ...form, [key]: value });
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 backdrop-blur-sm">
      <div className="w-full max-w-md p-6 bg-slate-800 rounded-2xl shadow-xl">
        <h3 className="mb-4 text-lg font-bold text-white">Game Settings</h3>

        <div className="grid grid-cols-2 gap-4">
          <label className="flex flex-col gap-1">
            <span className="text-sm text-slate-300">Starting Money</span>
            <input
              type="number"
              min={0}
              value={Number(form.startingMoney)}
              onChange={(e) => update("startingMoney", Number(e.target.value))}
              className="w-full px-3 py-2 text-white bg-slate-700 border border-slate-600 rounded focus:outline-none focus:border-brand-primary"
            />
          </label>

  startingMoney: number;

  winModifier: number;   // e.g., 0.35 for +35%, -0.15 for -15%
  highStakesMode: boolean;

export function getBalanceModifiers(startingMoney: number): { loss: number; win: number; highStakes: boolean } {
  if (startingMoney >= 500_000) {
    return { loss: 0.5, win: -0.35, highStakes: true }; // $500k-$1M: Lose +50%, Win -35%

    return { loss: 0.25, win: -0.15, highStakes: true }; // $100k-$499k: Lose +25%, Win -15%
  } else if (startingMoney < 1_000) {
    return { loss: -0.5, win: 0.35, highStakes: false }; // $100-$999: Lose -50%, Win +35%
  }
  return { loss: 0, win: 0, highStakes: false }; // Normal range
}

export function getInitials(name: string): string {
  return name
    .split(" ")
    .map((n) => n[0])
    .join("")
    .toUpperCase()
    .slice(0, 2);
}
